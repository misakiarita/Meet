
 # バージョン 
  Ruby 3.0.1  
  Ruby on Rails 6.0.5

 # 就業Termの技術
  お気に入り機能  
  メッセージ機能
  devise

 # カリキュラム外の技術
  CanCanCan

 # 準備 

  $ git clone git@github.com:git@github.com:misakiarita/Meet.git
  $ cd Meet
  $ bundle install
  $ yarn install
  $ rails db:create && rails db:migrate
  $ rails s

 # アプリケーション実行
  $ rails s
